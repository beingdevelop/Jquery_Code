<script type="text/javascript">
    $(document).ready(function() {
        $('#pass').keyup(function() {//user jokhon key press korbe tokhon action perform hobe
            var len;
            len = $('#pass').val().length;//user je password ta dibe tar length ta ekhane calculate hobe
            //$('.first').text(len);
            if (len < 1) {
                $('first').text('');
                $('first').remove();
            } else if (len <= 4) {
                $('.first').text('weak');
            } else if (len <= 8) {
                $('.first').text('good');
            } else {
                $('.first').text('very strong');
            }
        });
        $('#pass').keydown(function() {
            var len;
            len = $('#pass').val().length;
            //$('.first').text(len);
            if (len < 1) {
                $('first').text('');
                
            } else if (len <= 4) {
                $('.first').text('');
            } else if (len <= 8) {
                $('.first').text('');
            } else {
                $('.first').text('');
            }
        });
    });

</script>
